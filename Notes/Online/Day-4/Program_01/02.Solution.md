# 🧠 Solution — Diagonal Rainwater Trapping

---

## Concepts Used

- 2D Grid Traversal
- Graph Reachability (Restricted Directions)
- BFS (Reverse Flow)
- Minimum Boundary Constraint Propagation

---

## A. Complete Solution Code (C++)

```cpp
#include <bits/stdc++.h>
using namespace std;

int trapWater(vector<vector<int>>& grid) {
    int n = grid.size();
    if (n == 0) return 0;
    int m = grid[0].size();

    vector<vector<int>> visited(n, vector<int>(m, 0));
    queue<pair<int,int>> q;

    // Push all boundary cells
    for (int i = 0; i < n; i++) {
        for (int j : {0, m-1}) {
            q.push({i, j});
            visited[i][j] = 1;
        }
    }
    for (int j = 0; j < m; j++) {
        for (int i : {0, n-1}) {
            if (!visited[i][j]) {
                q.push({i, j});
                visited[i][j] = 1;
            }
        }
    }

    // Directions: ↙ and ↘ reverse (we go upward)
    vector<pair<int,int>> dirs = {{-1, -1}, {-1, 1}};

    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();

        for (auto [dx, dy] : dirs) {
            int nx = x + dx;
            int ny = y + dy;

            if (nx >= 0 && nx < n && ny >= 0 && ny < m && !visited[nx][ny]) {
                if (grid[nx][ny] <= grid[x][y]) {
                    visited[nx][ny] = 1;
                    q.push({nx, ny});
                }
            }
        }
    }

    int water = 0;

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (!visited[i][j]) {
                int minBound = INT_MAX;

                for (auto [dx, dy] : vector<pair<int,int>>{{1,-1},{1,1}}) {
                    int ni = i + dx;
                    int nj = j + dy;

                    if (ni >= 0 && ni < n && nj >= 0 && nj < m) {
                        minBound = min(minBound, grid[ni][nj]);
                    }
                }

                if (minBound > grid[i][j]) {
                    water += (minBound - grid[i][j]);
                }
            }
        }
    }

    return water;
}
```
