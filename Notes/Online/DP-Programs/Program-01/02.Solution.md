```cpp
#include <iostream>
#include <vector>
#include <unordered_set>

using namespace std;

int countWays(int N, const unordered_set<int>& broken) {
    // Create a vector to store the number of ways to reach each step
    vector<int> dp(N + 1, 0);
    
    // Base case: There is one way to be at the starting point (step 0)
    dp[0] = 1;

    // Fill the dp array
    for (int i = 1; i <= N; ++i) {
        // If the current step is broken, skip it
        if (broken.count(i)) {
            dp[i] = 0;
        } else {
            // The number of ways to reach step i is the sum of ways to reach step i-1 and i-2
            dp[i] = dp[i - 1] + (i > 1 ? dp[i - 2] : 0);
        }
    }

    return dp[N];
}

int main() {
    int N, M;
    cin >> N >> M;

    unordered_set<int> broken;
    for (int i = 0; i < M; ++i) {
        int step;
        cin >> step;
        broken.insert(step);
    }

    cout << countWays(N, broken) << endl;

    return 0;
}
```
